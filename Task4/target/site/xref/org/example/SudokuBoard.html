3   public class SudokuBoard {
4       private SudokuField[] board;
5       private SudokuRow[] rows = new SudokuRow[9];
6       private SudokuColumn[] columns = new SudokuColumn[9];
7       private SudokuBox[][] boxes = new SudokuBox[3][3];
8       private SudokuSolver sudokuSolver;
9   
10      public SudokuBoard(SudokuSolver solver) {
11          board = new SudokuField[81];
12          this.sudokuSolver = solver;
13          initializeSections();
14          initializeBoard();
15      }
16  
17      private void initializeSections() {
18          for (int i = 0; i < 9; i++) {
19              rows[i] = new SudokuRow();
20              columns[i] = new SudokuColumn();
21              boxes[i / 3][i % 3] = new SudokuBox();
22          }
23      }
24  
25      private void initializeBoard() {
26          for (int i = 0; i < 81; i++) {
27              SudokuField field = new SudokuField();
28              board[i] = field;
29  
30              int row = i / 9;
31              int col = i % 9;
32              int boxRow = row / 3;
33              int boxCol = col / 3;
34  
35              rows[row].setField(col, field);
36              columns[col].setField(row, field);
37              boxes[boxRow][boxCol].setField(row % 3 * 3 + col % 3, field);
38          }
39      }
40  
41      public int get(int x, int y) {
42          return board[y * 9 + x].getFieldValue();
43      }
44  
45      public void set(int x, int y, int value) {
46          board[y * 9 + x].setFieldValue(value);
47      }
48  
49      public boolean checkBoard() {
50          for (int i = 0; i < 3; i++) {
51              for (int j = 0; j < 3; j++) {
52                  if (!boxes[i][j].verify()) {
53                      return false;
54                  }
55              }
56          }
57  
58          for (SudokuRow row : rows) {
59              if (!row.verify()) {
60                  return false;
61              }
62          }
63  
64          for (SudokuColumn column : columns) {
65              if (!column.verify()) {
66                  return false;
67              }
68          }
69  
70          return true;
71      }
72  
73  
74      public SudokuRow getRow(int y) {
75          return rows[y];
76      }
77  
78      public SudokuColumn getColumn(int x) {
79          return columns[x];
80      }
81  
82      public SudokuBox getBox(int x, int y) {
83          return boxes[y / 3][x / 3];
84      }
85  
86      public boolean solveGame() {
87          return sudokuSolver.solve(this);
88      }
89  }
